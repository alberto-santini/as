227  inline TSPLIBInput read_tsplib_file(std::string tsplib_file) {
228  std::ifstream ifs(tsplib_file);
229 
230  if(ifs.fail()) {
231  throw std::runtime_error("Cannot read from file: " + tsplib_file);
232  }
233 
234  TSPLIBInput tsp;
235 
236  const std::regex specification_regex("([A-Z_]+)\\s*:\\s*(.+)");
237  std::string line;
238  std::size_t line_number = 0u;
239 
240  // We start reading the specification part, which must come first and
241  // before the data part. Data in the specification part must be in the
242  // format specified by specification_regex.
243  while(std::getline(ifs, line)) {
244  ++line_number;
245 
246  std::smatch match;
247 
248  if(std::regex_match(line, match, specification_regex)) {
249  // We want the regex to match fully.
250  // match[0] will contain the whole string;
251  // match[1] will contain the first parenthesised match;
252  // match[2] will contain the second parenthesised match.
253  if(match.size() != 3u) {
254  std::cerr << console::error << "Invalid format for a specification line. The line is:\n";
255  std::cerr << "\t" << line << "\n";
256  throw std::runtime_error("Invalid specification line on line number " + std::to_string(line_number));
257  }
258 
259  tsp.add_specification(match[1].str(), match[2].str());
260  } else {
261  // The specification part is probably over, we break and
262  // move to the data part.
263  break;
264  }
265  }
266 
267  const std::regex data_label_regex("^\\s*([A-Z_]+)\\s*$");
268  const std::string eof_marker = "EOF";
269  std::string current_label = "";
270 
271  // We start the data part. The first line of it has already been read,
272  // and used to break the previous loop. Therefore, we now switch the
273  // loop structure from a while-do to a do-while. The labels for the data
274  // part must be in the format specified by data_label_regex.
275  do {
276  if(line == eof_marker) { break; }
277 
278  std::smatch match;
279 
280  if(std::regex_match(line, match, data_label_regex)) {
281  // We want the regex to match fully.
282  // match[0] will contain the whole string;
283  // match[1] will contain the first parenthesised match.
284  if(match.size() != 2u) {
285  std::cerr << console::error << "Invalid format for a data label line. The line is:\n";
286  std::cerr << "\t" << line << "\n";
287  throw std::runtime_error("Invalid data label line on line number " + std::to_string(line_number));
288  }
289 
290  current_label = match[1].str();
291  tsp.add_data_block(current_label);
292  } else {
293  if(current_label == "") {
294  std::cerr << console::error << "Data line without a parent label. The line is:\n";
295  std::cerr << "\t" << line << "\n";
296  throw std::runtime_error("Invalid data line on line number " + std::to_string(line_number));
297  }
298 
299  tsp.add_data_block_elements(current_label, line);
300  }
301 
302  ++line_number;
303  } while(std::getline(ifs, line));
304 
305  return tsp;
306  }

338  inline std::function<float(float,float,float,float)> get_distance_function(std::string f_name) {
339  // Distances must be computed according to a specific formula
340  // for the results to be comparable with other methods in the
341  // literature. These formulas might not always seem the most
342  // intuitive...
343  if(f_name == "EUC_2D") {
344  return [] (float x1, float y1, float x2, float y2) -> float {
345  const auto xd = x1 - x2;
346  const auto yd = y1 - y2;
347  const auto d = std::sqrt(std::pow(xd, 2.0f) + std::pow(yd, 2.0f));
348 
349  return std::round(d); // Notice the rounding here!
350  };
351  } else if(f_name == "CEIL_2D") {
352  return [] (float x1, float y1, float x2, float y2) -> float {
353  const auto xd = x1 - x2;
354  const auto yd = y1 - y2;
355  const auto d = std::sqrt(std::pow(xd, 2.0f) + std::pow(yd, 2.0f));
356 
357  return std::ceil(d); // Notice the ceiling here!
358  };
359  } else if(f_name == "GEO") {
360  return [] (float x1, float y1, float x2, float y2) -> float {
361  const auto lat1 = latlon(x1), lon1 = latlon(y1);
362  const auto lat2 = latlon(x2), lon2 = latlon(y2);
363  const auto q1 = std::cos(lon1 - lon2);
364  const auto q2 = std::cos(lat1 - lat2);
365  const auto q3 = std::cos(lat1 + lat2);
366  const auto q = (1.0f + q1) * q2 - (1.0f - q1) * q3;
367  const auto qa = std::acos(0.5f * q);
368  const auto dist = tsp_earth_radius * qa + 1.0f;
369 
370  return dist;
371  };
372  } else if(f_name == "ATT") {
373  return [] (float x1, float y1, float x2, float y2) -> float {
374  const auto xd = x1 - x2;
375  const auto yd = y1 - y2;
376  const auto r = std::sqrt((std::pow(xd, 2.0f) + std::pow(yd, 2.0f)) / 10.0f);
377  const auto t = std::trunc(r);
378 
379  if(t < r) {
380  return t + 1;
381  } else {
382  return t;
383  }
384  };
385  } else {
386  throw std::domain_error("Distance function not supported: " + f_name);
387  }
388  }

404  inline std::pair<TwoDimPoint, TwoDimPoint> get_circle_intersection(
405  float dist_centre_1_point,
406  float dist_centre_2_point,
407  float dist_centre_1_centre_2
408  ) {
409  if(dist_centre_1_point + dist_centre_2_point <= dist_centre_1_centre_2) {
410  // The distances violate the triangle inequality: we have to
411  // consider one as true, and adjust the other one. In other words,
412  // we impose that the triangle inequality is satisfied with
413  // equality, that is, the three points are aligned. So there
414  // will only be one intersection, lying on the x axis.
415  const auto adj_x = dist_centre_1_centre_2 - dist_centre_2_point;
416  return std::make_pair(TwoDimPoint{adj_x, 0.0f}, TwoDimPoint{adj_x, 0.0f});
417  }
418 
419  const float delta = std::pow(dist_centre_1_point, 2.0f) -
420  std::pow(dist_centre_2_point, 2.0f) +
421  std::pow(dist_centre_1_centre_2, 2.0f);
422 
423  const float beta = 4.0f *
424  std::pow(dist_centre_1_point, 2.0f) *
425  std::pow(dist_centre_1_centre_2, 2.0f) -
426  std::pow(delta, 2.0f);
427 
428  const float x = delta / (2.0f * dist_centre_1_centre_2);
429  const float y = std::sqrt(std::abs(beta)) / (2.0f * dist_centre_1_centre_2);
430 
431  return std::make_pair(TwoDimPoint{x, y}, TwoDimPoint{x, -y});
432  }

633  void set_coordinates_geo(bool set_weights = false) {
634  // First get the coordinates as if they were
635  // Euclidean (x,y) coordinates, i.e. by just using
636  // the raw values given in the instance file.
637  set_coordinates_euclidean();
638 
639  // Then calculate the weights with these coordinates,
640  // using the GEO function.
641  if(set_weights) {
642  set_weights_from_coordinates();
643  }
644 
645  // Then project the coords using an Azimuthal Equidistant Projection
646  // centred on the depot. For more information see:
647  // http://mathworld.wolfram.com/AzimuthalEquidistantProjection.html
648  const auto eucl_coordinates = coordinates;
649  const auto centre_lat = detail::latlon(eucl_coordinates[0].x);
650  const auto centre_lon = detail::latlon(eucl_coordinates[0].y);
651 
652  coordinates[0] = {0.0f, 0.0f};
653 
654  for(auto i = 1u; i < n_vertices; ++i) {
655  const auto v_lat = detail::latlon(eucl_coordinates[i].x);
656  const auto v_lon = detail::latlon(eucl_coordinates[i].y);
657  const auto cos_c = std::sin(centre_lat) * std::sin(v_lat) +
658  std::cos(centre_lat) * std::cos(v_lat) * std::cos(v_lon - centre_lon);
659  const auto c = std::acos(cos_c);
660  const auto k = c / std::sin(c);
661  const auto x = detail::tsp_earth_radius * k *
662  (std::cos(centre_lat) * std::sin(v_lat) - std::sin(centre_lat) * std::cos(v_lat) * std::cos(v_lon - centre_lon));
663  const auto y = detail::tsp_earth_radius * k * std::cos(v_lat) * std::sin(v_lon - centre_lon);
664 
665  coordinates[i] = {x, y};
666  }
667  }

674  void set_coordinates_euclidean() {
675  const auto coords = tsp.get_data("NODE_COORD_SECTION");
676  coordinates.resize(n_vertices);
677 
678  // Coordinates come in triplets:
679  // - The first number is the vertex id
680  // - The second number is the x coordinate
681  // - The third number is the y coordinate
682  // Here we assume the vertex id is always sequential.
683  for(auto i = 0u; i < coords.size(); i += 3) {
684  std::size_t vertex_id = static_cast<std::size_t>(coords[i]) - 1;
685 
686  if(vertex_id != i / 3) {
687  throw std::logic_error("Node coordinates not given in sequential order");
688  }
689 
690  coordinates[vertex_id] = {coords[i+1], coords[i+2]};
691  }
692  }

694  void set_weights_from_coordinates() {
695  const auto dist_f = detail::get_distance_function(tsp.get_specification<std::string>("EDGE_WEIGHT_TYPE"));
696  distances.resize(n_vertices, std::vector<float>(n_vertices, 0.0f));
697 
698  for(auto i = 0u; i < n_vertices; ++i) {
699  for(auto j = i + 1; j < n_vertices; ++j) {
700  const auto dist = dist_f(
701  coordinates[i].x, coordinates[i].y,

703  );
704  distances[i][j] = dist;

710  void reverse_engineer_coordinates() {
711  coordinates.resize(n_vertices);
712 
713  assert(n_vertices >= 3u);
714 
715  // We place the first point on the origin.
716  coordinates[0] = {0.0f, 0.0f};
717 
718  // We place the second point on the positive x-semiaxis.
719  coordinates[1] = {distances[0][1], 0.0f};
720 
721  // We place the third point on the positive x-halfplane.
722  coordinates[2] = detail::get_circle_intersection(distances[0][2], distances[1][2], distances[0][1]).first;
723 
724  // We place all other points in the position that better matches their distance
725  // with the third point.
726  for(auto i = 3u; i < n_vertices; ++i) {
727  const auto positions = detail::get_circle_intersection(
728  distances[0][i], distances[1][i], distances[0][1]
729  );
730 
731  const auto distance_first = detail::euclidean_dist(coordinates[2], positions.first);
732  const auto distance_second = detail::euclidean_dist(coordinates[2], positions.second);
733 
734  if(std::abs(distance_first - distances[2][i]) < std::abs(distance_second - distances[2][i])) {
735  coordinates[i] = positions.first;
736  } else {
737  coordinates[i] = positions.second;
738  }
739  }
740  }
741  };
742  }
