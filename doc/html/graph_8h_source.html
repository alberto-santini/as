188

190  assert(

192  vertices.begin(), vertices.end()

194  );
195 
196  // Get the (ordered) set of all vertices in the graph.
197  std::vector<typename boost::graph_traits<BoostGraph>::vertex_descriptor> all_vertices(boost::num_vertices(graph));
198  std::iota(all_vertices.begin(), all_vertices.end(), typename boost::graph_traits<BoostGraph>::vertex_descriptor{});
199 
200  // Order the set of vertices passed by the user.
201  std::sort(vertices.begin(), vertices.end());
202 
203  // Return the set difference between the two.
204  std::vector<typename boost::graph_traits<BoostGraph>::vertex_descriptor> difference;
205  std::set_difference(all_vertices.begin(), all_vertices.end(),
206  vertices.begin(), vertices.end(),
